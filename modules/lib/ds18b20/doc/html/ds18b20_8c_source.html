153

155  if (scratchpad.temp_lsb & 0x08)
156  temp_fract += 5000;
157  if (scratchpad.temp_lsb & 0x04)
158  temp_fract += 2500;
159  if (scratchpad.temp_lsb & 0x02)
160  temp_fract += 1250;
161  if (scratchpad.temp_lsb & 0x01)
162  temp_fract += 625;
163 
164  *result = temp_sign * ((int32_t)temp_int * 10000 + temp_fract);
165  }
