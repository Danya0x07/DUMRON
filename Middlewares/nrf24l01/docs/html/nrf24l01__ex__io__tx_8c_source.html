nrf24l01_ex_io_tx.c
1 /*
2  * Этот пример работы с передатчиком показывает, как можно организовать
3  * двухсторонний обмен на NRF24L01/+. Данные от приёмника принимаются
4  * передатчиком вместе с пакетами автоподтверждения.
5  */
6 
7 #include <avr/io.h>
8 #include <util/delay.h>
9 #include <stdlib.h>
10 
11 #include <nrf24l01.h>
12 
13 // вспомогательные функции для демонстрации
14 extern void uart_init(void);
15 extern void uart_logs(const char *);
16 extern void uart_logi(int);
17 extern void setup_gpio(void);
18 extern void setup_spi(void);
19 extern uint8_t transceiver_irq_pin_is_high(void);
20 
21 struct tx_payload { // структура полезной нагрузки, отправляемой передатчиком
22  uint8_t data;
23  uint8_t other_data;
24 } out_buffer;
25 
26 struct ack_payload { // структура полезной нагрузки, принимаемой передатчиком
27  uint8_t data;
28  uint8_t other_data;
29 } in_buffer;
30 
31 void print_ack_pld(struct ack_payload *pld)
32 {
33  uart_logs("data: ");
34  uart_logi(pld->data);
35  uart_logs("\tother_data: ");
36  uart_logi(pld->other_data);
37  uart_logs("\n");
38 }

40 int main(void)
41 {
42  setup_gpio();
43  setup_spi();
44  uart_init();
45 
46  // сочиняем конфигурацию передатчика
47  struct nrf24l01_tx_config config = {
48  // пусть будет такой 3х-байтовый адрес
49  .address = (uint8_t [3]){0x42, 0x42, 0x42},
50  .addr_size = NRF24L01_ADDRS_3BYTE,
51 
52  // непринципиальные для данного примера настройки
53  .crc_mode = NRF24L01_CRC_1BYTE,
54  .datarate = NRF24L01_DATARATE_1MBPS,
55  .power = NRF24L01_POWER_N6DBM,
56  .retr_delay = NRF24L01_RETR_DELAY_1000US,
57  .retr_count = NRF24L01_RETR_COUNT_5,
58  .rf_channel = 42, // такой же как у приёмника
59 
60  // выбираем режим для приёма автоподтверждений с полезной нагрузкой
61  .mode = NRF24L01_TX_MODE_ACK_PAYLOAD,
62 
63  // включим все прерывания
64  .en_irq = NRF24L01_IRQ_ALL
65  };
66 
67  // на всякий случай ждём стабилизации трансивера после подачи питания
68  _delay_ms(NRF24L01_PWR_ON_DELAY_MS);
69 
70  if (nrf24l01_tx_configure(&config) < 0) {
71  /*
72  * Обрабатываем ошибку инициализации.
73  * Возможно она вызвана плохим контактом модуля с макетной платой,
74  * если таковая используется.
75  */
76  uart_logs("nrf24l01 init error\n");
77  }
78 
79 #if (NRF24L01_PLUS == 0) // только для модификации без +
80  nrf24l01_activate();
81 #endif
82 
83  uint8_t irq;
84 
85  // пусть начальные отправляемые данные будут такими
86  out_buffer.data = 0;
87  out_buffer.other_data = 255;
88 
89  for (;;) {
90  // записываем полезную нагрузку
91  nrf24l01_tx_write_pld(&out_buffer, sizeof(struct tx_payload));
92 
93  // отправляем
94  nrf24l01_tx_transmit();
95 
96  // ждём прерывание (спад напряжения на пине IRQ трансивера)
97  // правильнее было бы использовать внешнее прерывание МК
98  while (transceiver_irq_pin_is_high())
99  ;
100 
101  // получаем маску произошедших прерываний
102  irq = nrf24l01_get_interrupts();
103 
104  if (irq & NRF24L01_IRQ_TX_DS) { // если получено автоподтверждение
105  uart_logs("sent successfully\n");
106  nrf24l01_clear_interrupts(NRF24L01_IRQ_TX_DS);
107  }
108  if (irq & NRF24L01_IRQ_MAX_RT) { // если кончились попытки отправки
109  uart_logs("failed to send\n");
110 
111  // если заполнили очередь отправки, чистим
112  if (nrf24l01_full_tx_fifo())
113  nrf24l01_flush_tx_fifo();
114 
115  nrf24l01_clear_interrupts(NRF24L01_IRQ_MAX_RT);
116  }
117  if (irq & NRF24L01_IRQ_RX_DR) { // если прилетели данные с приёмника
118  do {
119  /*
120  * Поскольку используется динамический размер полезной нагрузки,
121  * его нужно считывать и проверять на валидность.
122  */
123  if (nrf24l01_read_pld_size() != sizeof(struct ack_payload)) {
124  // чистим входную очередь
125  nrf24l01_flush_rx_fifo();
126  nrf24l01_clear_interrupts(NRF24L01_IRQ_RX_DR);
127  uart_logs("incorrect ack pld size\n");
128  break;
129  }
130 
131  // считываем полезную нагрузку и сбрасываем прерывание
132  nrf24l01_read_pld(&in_buffer, sizeof(struct ack_payload));
133  nrf24l01_clear_interrupts(NRF24L01_IRQ_RX_DR);
134 
135  // печатаем принятые данные
136  uart_logs("ack payload recieved: ");
137  print_ack_pld(&in_buffer);
138  } while (nrf24l01_data_in_rx_fifo()); // пока есть данные в очереди
139  }
140 
141  // изменяем отправляемые данные для наглядности
142  out_buffer.data++;
143  out_buffer.other_data--;
144 
145  _delay_ms(200);
146  }
147 }
