35 
36  if (nrf24l01_tx_configure_default() < 0) {

41  */
42  uart_logs("nrf24l01 init error\n");
43  }
44 
45  uint8_t irq;

47 
48  for (;;) {
49  // записываем полезную нагрузку в очередь
50  nrf24l01_tx_write_pld(simple_msg, sizeof(simple_msg));
51 
52  // отправляем
53  nrf24l01_tx_transmit();
54 
55  // ждём прерывание (спад напряжения на пине IRQ трансивера)
56  // правильнее было бы использовать внешнее прерывание МК
57  while (transceiver_irq_pin_is_high())
58  ;
59 
60  // получаем маску произошедших прерываний
61  irq = nrf24l01_get_interrupts();
62 
63  if (irq & NRF24L01_IRQ_TX_DS) { // если получено автоподтверждение
64  uart_logs("sent successfully\n");
65  nrf24l01_clear_interrupts(NRF24L01_IRQ_TX_DS);
66  }
67  if (irq & NRF24L01_IRQ_MAX_RT) { // если кончились попытки отправки
68  uart_logs("failed to send\n");
69 
70  // если заполнили очередь отправки, чистим
71  if (nrf24l01_full_tx_fifo())
72  nrf24l01_flush_tx_fifo();
73 
74  nrf24l01_clear_interrupts(NRF24L01_IRQ_MAX_RT);
75  }
76 
77  // читаем и печатаем статистику радиообмена
78  nrf24l01_tx_get_statistics(&lost, &retr);
79 
80  uart_logs("lost: ");
81  uart_logi(lost);
82  uart_logs(" retransmits: ");
83  uart_logi(retr);
84  uart_logs("\n");
85 
86  _delay_ms(200);
87  }
88 }
