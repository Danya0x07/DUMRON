1 /*
2  * Этот пример работы с приёмником показывает, как можно организовать
3  * двухсторонний обмен на NRF24L01/+. Данные с приёмника отправляются
4  * передатчику вместе с пакетами автоподтверждения.
5  */

21 struct tx_payload { // структура полезной нагрузки, принимаемой от передатчика
22  uint8_t data;
23  uint8_t other_data;
24 } in_buffer;

27  uint8_t data;
28  uint8_t other_data;
29 } out_buffer;

32 {
33  uart_logs("data: ");

40 int main(void)
41 {
42  setup_gpio();
43  setup_spi();
44  uart_init();
45 
46  // сочиняем конфигурацию приёмника
47  struct nrf24l01_rx_config config = {
48  // пусть будет 3х-байтовый адрес, (как и у передатчика)
49  .addr_size = NRF24L01_ADDRS_3BYTE,
50 
51  // непринципиальные для данного примера настройки
52  .crc_mode = NRF24L01_CRC_1BYTE,
53  .datarate = NRF24L01_DATARATE_1MBPS,
54  .rf_channel = 42, // такой же как у передатчика
55 
56  // выбираем режим для отправки полезной нагрузки с автоподтверждениями
57  .mode = NRF24L01_RX_MODE_DPL_ACK_PAYLOAD,
58 
59  // включим прерывание по приёму пакета
60  .en_irq = NRF24L01_IRQ_RX_DR
61  };
62 
63  // сочиняем конфигурацию входящего соединения, например 0-го.
64  struct nrf24l01_pipe_config pipe0_conf = {
65  // т.к. выбрано соединение 0, адрес задаётся массивом, размер которого
66  // указан выше в конфигурации приёмника
67  .address.array = (uint8_t [3]){0x42, 0x42, 0x42},
68 
69  // выбираем режим для отправки полезной нагрузки с автоподтверждениями
70  .mode = NRF24L01_PIPE_MODE_ACK_DPL
71  /*
72  * Поскольку используется динамическая длина полезной нагрузки,
73  * поле pld_size инициализировать не нужно.
74  */
75  };
76 
77  // на всякий случай ждём стабилизации трансивера после подачи питания
78  _delay_ms(NRF24L01_PWR_ON_DELAY_MS);
79 
80  if (nrf24l01_rx_configure(&config) < 0) {
81  /*
82  * Обрабатываем ошибку инициализации.
83  * Возможно она вызвана плохим контактом модуля с макетной платой,
84  * если таковая используется.
85  */
86  uart_logs("nrf24l01 init error\n");
87  }
88 
89 #if (NRF24L01_PLUS == 0) // только для модификации без +
90  nrf24l01_activate();
91 #endif
92 
93  // настраиваем входящее соединение
94  nrf24l01_rx_setup_pipe(NRF24L01_PIPE0, &pipe0_conf);
95 
96  uint8_t irq;
97 
98  // пусть начальные отправляемые данные будут такими
99  out_buffer.data = 0;
100  out_buffer.other_data = 255;
101 
102  // начинаем слушать эфир
103  nrf24l01_rx_start_listening();
104 
105  for (;;) {
106  // если заполнили очередь отправки, чистим
107  if (nrf24l01_full_tx_fifo())
108  nrf24l01_flush_tx_fifo();
109 
110  // записываем в очередь полезную нагрузку для отправки вместе с
111  // автоподтверждением для 0-го соединения.
112  nrf24l01_rx_write_ack_pld(NRF24L01_PIPE0, &out_buffer,
113  sizeof(struct ack_payload));
114 
115  // ждём прерывание (спад напряжения на пине IRQ трансивера)
116  // правильнее было бы использовать внешнее прерывание МК
117  while (transceiver_irq_pin_is_high())
118  ;
119 
120  // если прилетели данные с передатчика
121  if (nrf24l01_get_interrupts() & NRF24L01_IRQ_RX_DR) {
122  do {
123  /*
124  * Поскольку используется динамический размер полезной нагрузки,
125  * его нужно считываь и проверять на валидность.
126  */
127  if (nrf24l01_read_pld_size() != sizeof(struct tx_payload)) {
128  // чистим входную очередь
129  nrf24l01_flush_rx_fifo();
130  nrf24l01_clear_interrupts(NRF24L01_IRQ_RX_DR);
131  uart_logs("incorrect ack pld size\n");
132  break;
133  }
134 
135  // считываем полезную нагрузку и сбрасываем прерывание
136  nrf24l01_read_pld(&in_buffer, sizeof(struct tx_payload));
137  nrf24l01_clear_interrupts(NRF24L01_IRQ_RX_DR);
138 
139  // печатаем принятые данные
140  uart_logs("payload recieved: ");
141  print_tx_pld(&in_buffer);
142  } while (nrf24l01_data_in_rx_fifo()); // пока есть данные в очереди
143  }
144 
145  // изменяем отправляемые данные для наглядности
146  out_buffer.data++;
147  out_buffer.other_data--;
148  }
149 }
