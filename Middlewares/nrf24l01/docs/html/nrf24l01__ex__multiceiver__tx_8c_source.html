4  * последовательной смены конфигураций передатчика.
5  */

18 extern void setup_spi(void);

20 struct p0_payload { // формат полезной нагрузки для 0-го соединения приёмника
21  uint8_t data0;
22 } pld0 = {0};
23 
24 struct p1_payload { // формат полезной нагрузки для 1-го соединения приёмника
25  uint8_t data0;
26  uint8_t data1;
27 } pld1 = {0};

29 struct p2_payload { // формат полезной нагрузки для 2-го соединения приёмника
30  uint8_t data0;
31  uint8_t data1;
32  uint8_t data2;
33 } pld2 = {0};

35 int main(void)
36 {
37  setup_gpio();
38  setup_spi();
39  uart_init();
40 
41  // адреса 3-х соединений приёмника
42  uint8_t address0[4] = {0x24, 0x24, 0x24, 0x24};
43  uint8_t address1[4] = {0x33, 0x33, 0x33, 0x33};
44  uint8_t address2[4] = {0x99, 0x33, 0x33, 0x33};
45 
46  // 3 конфигурации передатчика, каждая настроена на одно соединение приёмника
47 
48  struct nrf24l01_tx_config config0 = {
49  .address = address0,
50  .addr_size = NRF24L01_ADDRS_4BYTE,
51 
52  // непринципиальные для данного примера настройки
53  .crc_mode = NRF24L01_CRC_1BYTE,
54  .datarate = NRF24L01_DATARATE_2MBPS,
55  .power = NRF24L01_POWER_N6DBM,
56  .retr_delay = NRF24L01_RETR_DELAY_1000US,
57  .retr_count = NRF24L01_RETR_COUNT_4,
58  .rf_channel = 42, // такой же как у приёмника
59 
60  // выбираем режим для приёма автоподтверждений
61  .mode = NRF24L01_TX_MODE_ACK,
62 
63  // в этом примере не потребуется отслеживать логический уровень пина IRQ
64  .en_irq = 0
65  };
66 
67  // некоторые настройки вполне могут отличаться, в данном случае, для 1-го
68  // соединения приёмник настроен не высылать автоподтверждения
69  struct nrf24l01_tx_config config1 = config0;
70  config1.address = address1;
71  config1.mode = NRF24L01_TX_MODE_NOACK;
72 
73  struct nrf24l01_tx_config config2 = config0;
74  config2.address = address2;
75 
76  /*
77  * Механизм смены конфигураций: каждому "виртуальному" передатчику
78  * соответствует своя конфигурация, полезная нагрузка и её размер.
79  * Текущая конфигурация определяется индексом.
80  */
81  struct nrf24l01_tx_config *configs[3] = {&config0, &config1, &config2};
82  void *payloads[3] = {&pld0, &pld1, &pld2};
83  uint8_t pld_sizes[3] = {sizeof(pld0), sizeof(pld1), sizeof(pld2)};
84  int index = 0;
85 
86  // на всякий случай ждём стабилизации трансивера после подачи питания
87  _delay_ms(NRF24L01_PWR_ON_DELAY_MS);
88 
89  for (;;) {
90  // настраиваем передатчик
91  if (nrf24l01_tx_configure(configs[index]) < 0) {
92  uart_logs("nrf24l01 config error\n");
93  }
94 
95  // записываем полезную нагрузку
96  nrf24l01_tx_write_pld(payloads[index], pld_sizes[index]);
97 
98  // отправляем
99  nrf24l01_tx_transmit();
100 
101  // если кончились повторные попытки отправки
102  if (nrf24l01_get_interrupts() & NRF24L01_IRQ_MAX_RT) {
103  uart_logs("failed to send packet for pipe ");
104  uart_logi(index);
105  uart_logs("\n");
106 
107  // если заполнили очередь отправки, чистим
108  if (nrf24l01_full_tx_fifo())
109  nrf24l01_flush_tx_fifo();
110 
111  nrf24l01_clear_interrupts(NRF24L01_IRQ_MAX_RT);
112  }
113 
114  if (++index > 2) // индекс работает как циклический счётчик
115  index = 0;
116 
117  // изменяем отправляемые данные для наглядности
118  pld0.data0++;
119 
120  pld1.data0++;
121  pld1.data1++;
122 
123  pld2.data0++;
124  pld2.data1++;
125  pld2.data2++;
126 
127  _delay_ms(333);
128  }
129 }
