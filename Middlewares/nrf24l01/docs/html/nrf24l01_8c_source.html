142 int nrf24l01_tx_configure(struct nrf24l01_tx_config *config)
143 {
144  uint8_t features = 0;
145 
146  config->en_irq &= NRF24L01_IRQ_ALL;
147 
148  /*
149  * Запись 1 на место бита прерывания в регистре CONFIG выключает данное
150  * прерывание, поэтому пользовательскую маску нужно инвертировать.
151  */
152  nrf24l01_write_reg(REG_CONFIG, config->crc_mode |
153  (config->en_irq ^ NRF24L01_IRQ_ALL));
154 
155  nrf24l01_write_reg(REG_RF_SETUP, config->datarate | config->power);
156  nrf24l01_write_reg(REG_SETUP_RETR, config->retr_delay | config->retr_count);
157 
158  nrf24l01_write_reg(REG_SETUP_AW, config->addr_size);
159  /*
160  * Битовое представление ширины адреса определяется от 1 до 3,
161  * а фактический размер от 3 до 5 байт.
162  */
163  nrf24l01_write_buffer(CMD_W_REGISTER | REG_TX_ADDR, config->address,
164  config->addr_size + 2);
165 
166  if (config->mode >= NRF24L01_TX_MODE_ACK) {
167  /*
168  * Включаем и настраиваем входящее соединение 0 передатчика
169  * для приёма автоподтверждений.
170  */
171  nrf24l01_write_reg(REG_EN_RXADDR, 1 << NRF24L01_PIPE0);
172  nrf24l01_write_reg(REG_EN_AA, 1 << NRF24L01_PIPE0);
173  nrf24l01_write_buffer(CMD_W_REGISTER | REG_RX_ADDR_P0, config->address,
174  config->addr_size + 2);
175 
176  if (config->mode == NRF24L01_TX_MODE_ACK_PAYLOAD) {
177  features |=
178  (FEATURE_EN_DPL | FEATURE_EN_ACK_PAY | FEATURE_EN_DYN_ACK);
179  }
180  } else {
181  nrf24l01_write_reg(REG_EN_AA, 0);
182  nrf24l01_write_reg(REG_EN_RXADDR, 0);
183  }
184  nrf24l01_write_reg(REG_FEATURE, features);
185  nrf24l01_write_reg(REG_DYNPD, !!features << NRF24L01_PIPE0);
186 
187  nrf24l01_power_up();
188  nrf24l01_set_rf_channel(config->rf_channel);
189  nrf24l01_flush_tx_fifo();
190  nrf24l01_flush_rx_fifo();
191  nrf24l01_clear_interrupts(NRF24L01_IRQ_ALL);
192 
193  return nrf24l01_get_status() == STATUS_RESET_VALUE ? 0 : -1;
194 }

291 void nrf24l01_rx_setup_pipe(enum nrf24l01_pipe_number pipe_no,
292  struct nrf24l01_pipe_config *config)
293 {
294  if (config->pld_size > 32)
295  config->pld_size = 32;
296 
297  nrf24l01_write_bits(REG_EN_RXADDR, 1 << pipe_no, 1);
298  nrf24l01_write_reg(REG_RX_PW_P0 + pipe_no, config->pld_size);
299 
300  if (pipe_no == NRF24L01_PIPE0 || pipe_no == NRF24L01_PIPE1) {
301  uint8_t addr_size = nrf24l01_read_reg(REG_SETUP_AW) + 2;
302  nrf24l01_write_buffer(CMD_W_REGISTER | (REG_RX_ADDR_P0 + pipe_no),
303  config->address.array, addr_size);
304  } else {
305  nrf24l01_write_buffer(CMD_W_REGISTER | (REG_RX_ADDR_P0 + pipe_no),
306  &config->address.lsb, 1);
307  }
308  nrf24l01_write_bits(REG_EN_AA, 1 << pipe_no,
309  config->mode >= NRF24L01_PIPE_MODE_ACK);
310  nrf24l01_write_bits(REG_DYNPD, 1 << pipe_no,
311  config->mode == NRF24L01_PIPE_MODE_ACK_DPL);
312 }

420 bool nrf24l01_detect_signal(void)
421 {
422  const uint8_t config_backup = nrf24l01_read_reg(REG_CONFIG);
423 
424  /*
425  * Чтобы эту функцию можно было вызывать в режиме передатчика,
426  * сохраняем значение регистра настроек, а потом восстанавливаем его.
427  */
428  if ((config_backup & CONFIG_PRIM_RX) == 0)
429  nrf24l01_write_reg(REG_CONFIG, config_backup | CONFIG_PRIM_RX);
430 
431  _ce_high();
432  _delay_rpd();
433  _ce_low();
434 
435  if ((config_backup & CONFIG_PRIM_RX) == 0)
436  nrf24l01_write_reg(REG_CONFIG, config_backup);
437 
438  return nrf24l01_read_reg(REG_RPD_CD) & 0x01;
439 }

441 void nrf24l01_measure_noise(uint8_t *snapshot_buff,
442  uint8_t start_ch, uint8_t end_ch)
443 {
444  uint_fast8_t i, j;
445  const uint8_t config_backup = nrf24l01_read_reg(REG_CONFIG);
446 
447  if (end_ch > NRF24L01_CHANNELS - 1 || start_ch > end_ch)
448  return;
449 
450  for (i = 0; i < end_ch - start_ch + 1; i++)
451  snapshot_buff[i] = 0;
452 
453  /*
454  * Чтобы не происходило слишком частой записи по SPI в трансивер,
455  * сохраняем регистр настроек до цикла измерения.
456  */
457  if ((config_backup & CONFIG_PRIM_RX) == 0)
458  nrf24l01_write_reg(REG_CONFIG, config_backup | CONFIG_PRIM_RX);
459 
460  for (i = 0; i < 0xF; i++) {
461  for (j = start_ch; j <= end_ch; j++) {
462  nrf24l01_set_rf_channel(j);
463  if (nrf24l01_detect_signal())
464  snapshot_buff[j - start_ch]++;
465  }
466  }
467 
468  if ((config_backup & CONFIG_PRIM_RX) == 0)
469  nrf24l01_write_reg(REG_CONFIG, config_backup);
470 }

472 void nrf24l01_start_output_carrier(enum nrf24l01_power power, uint8_t channel)
473 {
474 #if (NRF24L01_PLUS == 1)
475  nrf24l01_power_up();
476  nrf24l01_write_bits(REG_CONFIG, CONFIG_PRIM_RX, 0);
477  nrf24l01_write_reg(REG_RF_SETUP, RF_SETUP_CONT_WAVE | RF_SETUP_PLL_LOCK |
478  power);
479  nrf24l01_set_rf_channel(channel);
480  _ce_high();
481 #else
482  uint_fast8_t i;
483 
484  nrf24l01_power_up();
485  nrf24l01_write_bits(REG_CONFIG, CONFIG_PRIM_RX, 0);
486  nrf24l01_write_reg(REG_EN_AA, 0);
487  nrf24l01_write_reg(REG_SETUP_RETR, 0);
488  nrf24l01_write_reg(REG_RF_SETUP, RF_SETUP_PLL_LOCK | power);
489  nrf24l01_write_reg(REG_SETUP_AW, NRF24L01_ADDRS_5BYTE);
490 
491  _spi_transfer_byte(CMD_W_REGISTER | REG_TX_ADDR);
492  for (i = 0; i < 5; i++)
493  _spi_transfer_byte(0xFF);
494 
495  _spi_transfer_byte(CMD_W_TX_PAYLOAD);
496  for (i = 0; i < 32; i++)
497  _spi_transfer_byte(0xFF);
498 
499  nrf24l01_write_bits(REG_CONFIG, CONFIG_EN_CRC, 0);
500  nrf24l01_set_rf_channel(channel);
501  nrf24l01_tx_transmit();
502  _delay_ms(1);
503  _ce_high();
