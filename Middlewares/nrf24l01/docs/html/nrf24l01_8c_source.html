142 int nrf24l01_tx_configure(struct nrf24l01_tx_config *config)
143 {
144  uint8_t features = 0;
145 
146  config->en_irq &= NRF24L01_IRQ_ALL;
147 
148  /*
149  * Запись 1 на место бита прерывания в регистре CONFIG выключает данное
150  * прерывание, поэтому пользовательскую маску нужно инвертировать.
151  */
152  nrf24l01_write_reg(REG_CONFIG, config->crc_mode |
153  (config->en_irq ^ NRF24L01_IRQ_ALL));
154 
155  nrf24l01_write_reg(REG_RF_SETUP, config->datarate | config->power);
156  nrf24l01_write_reg(REG_SETUP_RETR, config->retr_delay | config->retr_count);
157 
158  nrf24l01_write_reg(REG_SETUP_AW, config->addr_size);
159  /*
160  * Битовое представление ширины адреса определяется от 1 до 3,
161  * а фактический размер от 3 до 5 байт.
162  */
163  nrf24l01_write_buffer(CMD_W_REGISTER | REG_TX_ADDR, config->address,
164  config->addr_size + 2);
165 
166  if (config->mode >= NRF24L01_TX_MODE_ACK) {
167  /*
168  * Включаем и настраиваем входящее соединение 0 передатчика
169  * для приёма автоподтверждений.
170  */
171  nrf24l01_write_reg(REG_EN_RXADDR, 1 << NRF24L01_PIPE0);
172  nrf24l01_write_reg(REG_EN_AA, 1 << NRF24L01_PIPE0);
173  nrf24l01_write_buffer(CMD_W_REGISTER | REG_RX_ADDR_P0, config->address,
174  config->addr_size + 2);
175 
176  if (config->mode == NRF24L01_TX_MODE_ACK_PAYLOAD) {
177  features |=
178  (FEATURE_EN_DPL | FEATURE_EN_ACK_PAY | FEATURE_EN_DYN_ACK);
179  }
180  } else {
181  nrf24l01_write_reg(REG_EN_AA, 0);
182  nrf24l01_write_reg(REG_EN_RXADDR, 0);
183  }
184  nrf24l01_write_reg(REG_FEATURE, features);
185  nrf24l01_write_reg(REG_DYNPD, !!features << NRF24L01_PIPE0);
186 
187  nrf24l01_power_up();
188  nrf24l01_set_rf_channel(config->rf_channel);
189  nrf24l01_flush_tx_fifo();
190  nrf24l01_flush_rx_fifo();
191  nrf24l01_clear_interrupts(NRF24L01_IRQ_ALL);
192 
193  return nrf24l01_get_status() == STATUS_RESET_VALUE ? 0 : -1;
194 }

251 int nrf24l01_rx_configure(struct nrf24l01_rx_config *config)
252 {
253  uint8_t features = 0;
254 
255  config->en_irq &= NRF24L01_IRQ_ALL;
256 
257  nrf24l01_write_reg(REG_CONFIG, config->crc_mode | CONFIG_PRIM_RX |
258  (config->en_irq ^ NRF24L01_IRQ_ALL));
259 
260  nrf24l01_write_reg(REG_RF_SETUP, config->datarate);
261  nrf24l01_write_reg(REG_SETUP_AW, config->addr_size);
262 
263  if (config->mode >= NRF24L01_RX_MODE_DPL) {
264  features |= FEATURE_EN_DPL;
265  if (config->mode == NRF24L01_RX_MODE_DPL_ACK_PAYLOAD)
266  features |= FEATURE_EN_ACK_PAY | FEATURE_EN_DYN_ACK;
267  }
268  nrf24l01_write_reg(REG_FEATURE, features);
269 
270  nrf24l01_power_up();
271  nrf24l01_set_rf_channel(config->rf_channel);
272  nrf24l01_flush_tx_fifo();
273  nrf24l01_flush_rx_fifo();
274  nrf24l01_clear_interrupts(NRF24L01_IRQ_ALL);
275 
276  return nrf24l01_get_status() == STATUS_RESET_VALUE ? 0 : -1;
277 }

290 void nrf24l01_rx_setup_pipe(struct nrf24l01_pipe_config *config)
291 {
292  if (config->pld_size > 32)
293  config->pld_size = 32;
294 
295  nrf24l01_write_bits(REG_EN_RXADDR, 1 << config->number, 1);
296  nrf24l01_write_reg(REG_RX_PW_P0 + config->number, config->pld_size);
297 
298  if (config->number == NRF24L01_PIPE0 || config->number == NRF24L01_PIPE1) {
299  uint8_t addr_size = nrf24l01_read_reg(REG_SETUP_AW) + 2;
300  nrf24l01_write_buffer(CMD_W_REGISTER |
301  (REG_RX_ADDR_P0 + config->number),
302  config->address.array, addr_size);
303  } else {
304  nrf24l01_write_buffer(CMD_W_REGISTER |
305  (REG_RX_ADDR_P0 + config->number),
306  &config->address.lsb, 1);
307  }
308  nrf24l01_write_bits(REG_EN_AA, 1 << config->number,
309  config->mode >= NRF24L01_PIPE_MODE_ACK);
310  nrf24l01_write_bits(REG_DYNPD, 1 << config->number,
311  config->mode == NRF24L01_PIPE_MODE_ACK_DPL);
312 }

432 void nrf24l01_measure_noise(uint8_t *snapshot_buff,
433  uint8_t start_ch, uint8_t end_ch)
434 {
435  uint_fast8_t i, j;
436 
437  if (end_ch > NRF24L01_CHANNELS - 1 || start_ch > end_ch)
438  return;
439 
440  for (i = 0; i < end_ch - start_ch + 1; i++)
441  snapshot_buff[i] = 0;
442 
443  for (i = 0; i < 0xF; i++) {
444  for (j = start_ch; j <= end_ch; j++) {
445  nrf24l01_set_rf_channel(j);
446  if (nrf24l01_detect_signal())
447  snapshot_buff[j - start_ch]++;
448  }
449  }
450 }

453 {

458  power);
459  nrf24l01_set_rf_channel(channel);
460  _ce_high();
461 #else
462  uint_fast8_t i;
463 
464  nrf24l01_power_up();

466  nrf24l01_write_reg(REG_EN_AA, 0);
467  nrf24l01_write_reg(REG_SETUP_RETR, 0);
468  nrf24l01_write_reg(REG_RF_SETUP, RF_SETUP_PLL_LOCK | power);
469  nrf24l01_write_reg(REG_SETUP_AW, NRF24L01_ADDRS_5BYTE);
470 
471  _spi_transfer_byte(CMD_W_REGISTER | REG_TX_ADDR);
472  for (i = 0; i < 5; i++)

474

476  for (i = 0; i < 32; i++)
477  _spi_transfer_byte(0xFF);
478 
479  nrf24l01_write_bits(REG_CONFIG, CONFIG_EN_CRC, 0);
480  nrf24l01_set_rf_channel(channel);
481  nrf24l01_tx_transmit();

483  _ce_high();
